162  void gdalcubes_init() {
163  curl_global_init(CURL_GLOBAL_ALL);
164  GDALAllRegister();
165  GDALSetCacheMax(_gdal_cache_max);
166  CPLSetConfigOption("GDAL_PAM_ENABLED", "NO"); // avoid aux files for PNG tiles
167  curl_global_init(CURL_GLOBAL_ALL);
168  CPLSetConfigOption("GDAL_NUM_THREADS", std::to_string(_gdal_num_threads).c_str());
169  //srand(time(NULL)); // R will complain if calling srand...
170  CPLSetErrorHandler(CPLQuietErrorHandler);

172  // Add default locations where to look for collection format presets
173 
174  if (std::getenv("GDALCUBES_DATA_DIR") != NULL) {
175  if (filesystem::exists(std::getenv("GDALCUBES_DATA_DIR"))) {
176  config::instance()->add_collection_format_preset_dir(std::getenv("GDALCUBES_DATA_DIR"));

179  if (std::getenv("AllUsersProfile") != NULL) {
180  // Windows default location
181  std::string p = filesystem::join(filesystem::join(std::getenv("AllUsersProfile"), "gdalcubes"), "formats");
182  if (filesystem::exists(p)) {
183  config::instance()->add_collection_format_preset_dir(p);
184  }
185  }
186  if (std::getenv("HOME") != NULL) {
187  std::string p = filesystem::join(filesystem::join(std::getenv("HOME"), ".gdalcubes"), "formats");
188  if (filesystem::exists(p)) {
189  config::instance()->add_collection_format_preset_dir(p);
190  }
191  }
192  if (std::getenv("HOMEPATH") != NULL && getenv("HOMEDRIVE") != NULL) {
193  std::string p = filesystem::join(filesystem::join(std::getenv("HOMEPATH"), ".gdalcubes"), "formats");
