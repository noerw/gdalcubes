38 struct value_mask : public image_mask {
39  public:
40  value_mask(std::unordered_set<double> mask_values, bool invert = false, std::vector<uint8_t> bits = std::vector<uint8_t>()) : _mask_values(mask_values), _invert(invert), _bits(bits) {}
41  void apply(double *mask_buf, double *pixel_buf, uint32_t nb, uint32_t ny, uint32_t nx) override {
42  uint32_t bitmask = 0;
43  if (!_bits.empty()) {
44  for (uint8_t ib = 0; ib < _bits.size(); ++ib) {
45  bitmask += (uint32_t)std::pow(2.0, (double)_bits[ib]);
46  }
47  }
48  if (!_invert) {
49  for (uint32_t ixy = 0; ixy < ny * nx; ++ixy) {
50  if (!_bits.empty()) {
51  mask_buf[ixy] = (uint32_t)(mask_buf[ixy]) & bitmask;
52  }
53  if (_mask_values.count(mask_buf[ixy]) == 1) { // if mask has value
54  // set all bands to NAN
55  for (uint32_t ib = 0; ib < nb; ++ib) {
56  pixel_buf[ib * nx * ny + ixy] = NAN;
57  }
58  }
59  }
60  } else {
61  for (uint32_t ixy = 0; ixy < ny * nx; ++ixy) {
62  if (!_bits.empty()) {
63  mask_buf[ixy] = (uint32_t)(mask_buf[ixy]) & bitmask;
64  }
65  if (_mask_values.count(mask_buf[ixy]) == 0) {
66  // set all bands to NAN
67  for (uint32_t ib = 0; ib < nb; ++ib) {
68  pixel_buf[ib * nx * ny + ixy] = NAN;
69  }
70  }
71  }
72  }
73  }

94  void apply(double *mask_buf, double *pixel_buf, uint32_t nb, uint32_t ny, uint32_t nx) override {
95  uint32_t bitmask = 0;
96  if (!_bits.empty()) {
97  for (uint8_t ib = 0; ib < _bits.size(); ++ib) {
98  bitmask += (uint32_t)std::pow(2.0, (double)_bits[ib]);
99  }
100  }
101  if (!_invert) {
102  for (uint32_t ixy = 0; ixy < ny * nx; ++ixy) {
103  if (!_bits.empty()) {
104  mask_buf[ixy] = (uint32_t)(mask_buf[ixy]) & bitmask;
105  }
106  if (mask_buf[ixy] >= _min && mask_buf[ixy] <= _max) {
107  // set all bands to NAN
108  for (uint32_t ib = 0; ib < nb; ++ib) {
109  pixel_buf[ib * nx * ny + ixy] = NAN;
110  }
111  }
112  }
113  } else {
114  for (uint32_t ixy = 0; ixy < ny * nx; ++ixy) {
115  if (!_bits.empty()) {
116  mask_buf[ixy] = (uint32_t)(mask_buf[ixy]) & bitmask;
117  }
118  if (mask_buf[ixy] < _min || mask_buf[ixy] > _max) {
119  // set all bands to NAN
120  for (uint32_t ib = 0; ib < nb; ++ib) {
121  pixel_buf[ib * nx * ny + ixy] = NAN;
122  }
123  }
124  }
