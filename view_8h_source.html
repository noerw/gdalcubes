260  inline void dx(double x) {
261  _nx = (uint32_t)std::ceil((_win.right - _win.left) / x);
262  double exp_x = _nx * x - (_win.right - _win.left);
263  _win.right += exp_x / 2;
264  _win.left -= exp_x / 2;
265  if (std::fabs(exp_x) > std::numeric_limits<double>::epsilon()) {
266  GCBS_INFO("Size of the cube in x direction does not align with dx, extent will be enlarged by " +
267  std::to_string(exp_x / 2) + " at both sides.");
268  }
269  }

275  inline double dy() { return (_win.top - _win.bottom) / _ny; }

282  inline void dy(double x) {

349  uint32_t nt() {
350  if (_t1 == _t0) return 1;
351  duration d = (_t1 - _t0) + 1;
352  return (d % _dt == 0) ? d / _dt : (1 + (d / _dt));
353  }

359  void nt(uint32_t n) {
360  duration d = (_t1 - _t0) + 1;
361  duration dnew = dt();
362  if (dnew.dt_interval == 0) { // if dt has not been set
363  dnew.dt_unit = d.dt_unit;
364  // alternatively, a "reasonable" should be derived here
365  }
366  dnew.dt_interval = (int32_t)std::ceil((double)d.dt_interval / (double)n);
367  _dt = dnew;
368  if (d.dt_interval % n != 0) {
369  _t1 = _t0 + _dt * (n - 1);
370  GCBS_INFO(
371  "Temporal size of the cube does not align with nt, end date/time of the cube will be extended to " +
372  _t1.to_string() + ".");
373  }
374  //
375  // if (nt() == n - 1) { // in some cases (e.g. d == 9M, n==4), we must extend the temporal extent of the view
376  // _t1 = _t1 + dt();
377  // GCBS_WARN("Extent in t direction is indivisible by nt, end date/time will be set to " + _t1.to_string());

402  void dt(duration dt) {
403  //if (dt.dt_unit != _dt.dt_unit) {
404  _t0.unit() = dt.dt_unit;
405  _t1.unit() = dt.dt_unit;
406  //}
407  duration dtotal = _t1 - _t0; // + 1 if include end date2
408  dtotal.dt_interval += 1;
409  if (dtotal % dt != 0) {
410  duration end_duration; // end duration has one (day / month / unit) less than dt)
411  end_duration.dt_interval = dt.dt_interval - 1;
412  end_duration.dt_unit = dt.dt_unit;
413  _t1 = (_t0 + dt * (dtotal / dt)) + end_duration;
414  GCBS_INFO(
415  "Temporal size of the cube does not align with dt, end date/time of the cube will be extended to " +
416  _t1.to_string());
417  }
418  _dt = dt;
419  }
