113  // Find out, which bands are actually used per expression
114  for (uint16_t i = 0; i < _expr.size(); ++i) {
115  _band_usage.push_back(std::unordered_set<std::string>());
116  for (uint16_t ib = 0; ib < _in_cube->bands().count(); ++ib) {

122

124  _band_usage[i].insert(name);

126  }

131  if (_expr[i].find("it") != std::string::npos) _var_usage[i].insert("it");

136  if (_expr[i].find("t0") != std::string::npos) _var_usage[i].insert("t0");
