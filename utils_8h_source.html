50  static std::string generate_unique_filename(uint16_t n = 8, std::string prefix = "", std::string suffix = "") {
51  static std::mt19937 gen(time(NULL)); //Standard mersenne_twister_engine seeded with rd()
52  static const std::string LETTERS = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789";
53  static std::uniform_int_distribution<> dis(0, LETTERS.length() - 1);
54  static std::mutex mtx;
55  mtx.lock();
56  std::stringstream ss;
57  for (uint16_t i = 0; i < n; ++i) {
58  ss << LETTERS[dis(gen)];
59  }
60  std::string out = prefix + ss.str() + suffix;
61  mtx.unlock();
62  return out;
63  }
