441  chunkid_t find_chunk_that_contains(coords_st p) const {
442  uint32_t cumprod = 1;
443  chunkid_t id = 0;
444 
445  // 1. Convert map coordinates to view-based coordinates
446  coords_nd<uint32_t, 3> s = _st_ref->cube_coords(p);
447 
448  // 2. Find out which chunk contains the integer view coordinates
449  id += cumprod * (s[2] / _chunk_size[2]);
450  cumprod *= (uint32_t)std::ceil(((double)_st_ref->nx()) / ((double)_chunk_size[2]));
451  id += cumprod * (s[1] / _chunk_size[1]);
452  cumprod *= (uint32_t)std::ceil(((double)(_st_ref->ny()) / ((double)_chunk_size[1])));
453  id += cumprod * (s[0] / _chunk_size[0]);
454  //cumprod *= (uint32_t)std::ceil(((double)(_c->view().nt()) / ((double)_size_t);
455 
456  return id;
457  }

464  bounds_nd<uint32_t, 3> chunk_limits(chunk_coordinate_tyx c) const {
465  cube_coordinate_tyx out_vcoords_low;
466  cube_coordinate_tyx out_vcoords_high;
467 
468  out_vcoords_low[0] = c[0] * _chunk_size[0];
469  out_vcoords_low[1] = c[1] * _chunk_size[1];
470  out_vcoords_low[2] = c[2] * _chunk_size[2];
471  out_vcoords_high[0] = out_vcoords_low[0] + _chunk_size[0] - 1;
472  out_vcoords_high[1] = out_vcoords_low[1] + _chunk_size[1] - 1;
473  out_vcoords_high[2] = out_vcoords_low[2] + _chunk_size[2] - 1;
474 
475  // Shrink to view
476  if (out_vcoords_high[0] >= _st_ref->nt())
477  out_vcoords_high[0] = _st_ref->nt() - 1;
478  if (out_vcoords_low[0] >= _st_ref->nt())
479  out_vcoords_low[0] = _st_ref->nt() - 1;
480 
481  if (out_vcoords_high[1] >= _st_ref->ny())
482  out_vcoords_high[1] = _st_ref->ny() - 1;
483  if (out_vcoords_low[1] >= _st_ref->ny())
484  out_vcoords_low[1] = _st_ref->ny() - 1;
485 
486  if (out_vcoords_high[2] >= _st_ref->nx())
487  out_vcoords_high[2] = _st_ref->nx() - 1;
488  if (out_vcoords_low[2] >= _st_ref->nx())
489  out_vcoords_low[2] = _st_ref->nx() - 1;
490 
491  bounds_nd<uint32_t, 3> out;
492  out.low = out_vcoords_low;
493  out.high = out_vcoords_high;
494  return out;
495  };

502  bounds_nd<uint32_t, 3> chunk_limits(chunkid_t id) const {
503  coords_nd<uint32_t, 3> out_vcoords_low;
504  coords_nd<uint32_t, 3> out_vcoords_high;
505 
506  // Transform to global coordinates based on chunk id
507  uint32_t cumprod = 1;
508  int32_t n;
509 
510  n = (uint32_t)std::ceil(((double)(_st_ref->nx())) / ((double)_chunk_size[2]));
511  out_vcoords_low[2] = (id / cumprod) % n;
512  out_vcoords_low[2] *= _chunk_size[2];
513  out_vcoords_high[2] = out_vcoords_low[2] + _chunk_size[2] - 1;
514  cumprod *= n;
515 
516  n = (uint32_t)std::ceil(((double)(_st_ref->ny())) / ((double)_chunk_size[1]));
517  out_vcoords_low[1] = (id / cumprod) % n;
518  out_vcoords_low[1] *= _chunk_size[1];
519  out_vcoords_high[1] = out_vcoords_low[1] + _chunk_size[1] - 1;
520  cumprod *= n;
521 
522  n = (uint32_t)std::ceil(((double)(_st_ref->nt())) / ((double)_chunk_size[0]));
523  out_vcoords_low[0] = (id / cumprod) % n;
524  out_vcoords_low[0] *= _chunk_size[0];
525  out_vcoords_high[0] = out_vcoords_low[0] + _chunk_size[0] - 1;
526  cumprod *= n;
527 
528  // Shrink to view
529  if (out_vcoords_high[0] >= _st_ref->nt())
530  out_vcoords_high[0] = _st_ref->nt() - 1;
531  if (out_vcoords_low[0] >= _st_ref->nt())
532  out_vcoords_low[0] = _st_ref->nt() - 1;
533 
534  if (out_vcoords_high[1] >= _st_ref->ny())
535  out_vcoords_high[1] = _st_ref->ny() - 1;
536  if (out_vcoords_low[1] >= _st_ref->ny())
537  out_vcoords_low[1] = _st_ref->ny() - 1;
538 
539  if (out_vcoords_high[2] >= _st_ref->nx())
540  out_vcoords_high[2] = _st_ref->nx() - 1;
541  if (out_vcoords_low[2] >= _st_ref->nx())
542  out_vcoords_low[2] = _st_ref->nx() - 1;
543 
544  bounds_nd<uint32_t, 3> out;
545  out.low = out_vcoords_low;
546  out.high = out_vcoords_high;
547  return out;
548  };
