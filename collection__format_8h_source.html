58  static std::map<std::string, std::string> list_presets() {
59  std::map<std::string, std::string> out;
60 
61  std::vector<std::string> dirs = config::instance()->get_collection_format_preset_dirs();
62 
63  // do not use uint here because of descending iteration
64  for (int i = dirs.size() - 1; i >= 0; --i) {
65  if (!filesystem::exists(dirs[i])) {
66  continue;
67  }
68 
69  filesystem::iterate_directory(dirs[i], [&out](const std::string& p) {
70  if (filesystem::is_regular_file(p) && filesystem::extension(p) == "json") {
71  if (out.find(filesystem::stem(p)) == out.end())
72  out.insert(std::pair<std::string, std::string>(filesystem::stem(p), filesystem::make_absolute(p)));
73  }
74  });
75  }
76  return out;
77  }

85  void load_file(std::string filename) {
86  _j.clear();
87 
88  if ((!filesystem::exists(filename) && !filesystem::is_absolute(filename) && filesystem::parent(filename).empty()) || filesystem::is_directory(filename)) {
89  // simple filename without directories
90  GCBS_DEBUG("Couldn't find collection format '" + filename + "', looking for a preset with the same name");
91  std::map<std::string, std::string> preset_formats = list_presets();
92  if (preset_formats.find(filesystem::stem(filename)) != preset_formats.end()) {
93  filename = preset_formats[filesystem::stem(filename)];
94  GCBS_DEBUG("Found collection format preset at '" + filename + "'");
95  }
96  }
97  if (!filesystem::exists(filename) || filesystem::is_directory(filename))
98  throw std::string("ERROR in collection_format::load_file(): image collection format file does not exist.");
99 
100  std::ifstream i(filename);
101  i >> _j;
102  }
