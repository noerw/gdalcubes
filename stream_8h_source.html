60  stream_cube(std::shared_ptr<cube> in_cube, std::string cmd, bool file_streaming = false) : cube(std::make_shared<cube_st_reference>(*(in_cube->st_reference()))), _in_cube(in_cube), _cmd(cmd), _file_streaming(file_streaming), _keep_input_nt(false), _keep_input_ny(false), _keep_input_nx(false) { // it is important to duplicate st reference here, otherwise changes will affect input cube as well
61  // Test CMD and find out what size comes out.
62  cube_size_tyx tmp = _in_cube->chunk_size(0);
63  cube_size_btyx csize_in = {_in_cube->bands().count(), tmp[0], tmp[1], tmp[2]};
64 
65  // do not read original chunk data (which can be expensive) but simply stream a dummy chunk with proper size here
66  std::shared_ptr<chunk_data> dummy_chunk = std::make_shared<chunk_data>();
67  dummy_chunk->size(csize_in);
68  dummy_chunk->buf(std::calloc(csize_in[0] * csize_in[1] * csize_in[2] * csize_in[3], sizeof(double)));
69 
70  std::shared_ptr<chunk_data> c0;
71  if (_file_streaming) {
72  c0 = stream_chunk_file(dummy_chunk, 0);
73  } else {
74  c0 = stream_chunk_stdin(dummy_chunk, 0);
75  }
76 
77  for (uint16_t ib = 0; ib < c0->size()[0]; ++ib) {
78  band b("band" + std::to_string(ib + 1));
79  b.no_data_value = std::to_string(NAN);
80  b.type = "float64";
81  _bands.add(b);
82  }
83 
84  _chunk_size = {c0->size()[1], c0->size()[2], c0->size()[3]};
85 
86  // Make an optimistic guess
87  if (c0->size()[1] == 1) {
88  _st_ref->nt(in_cube->count_chunks_t());
89  } else if (c0->size()[1] == csize_in[1]) {
90  _keep_input_nt = true;
91  _st_ref->nt(in_cube->size()[1]);
92  } else
93  throw std::string("ERROR in stream_cube::stream_cube(): could not derive size of result cube");
94 
95  if (c0->size()[2] == 1) {
96  _st_ref->ny() = in_cube->count_chunks_y();
97  } else if (c0->size()[2] == csize_in[2]) {
98  _keep_input_ny = true;
99  _st_ref->ny() = in_cube->size()[2];
100  } else
101  throw std::string("ERROR in stream_cube::stream_cube(): could not derive size of result cube");
102 
103  if (c0->size()[3] == 1) {
104  _st_ref->nx() = in_cube->count_chunks_x();
105  } else if (c0->size()[3] == csize_in[3]) {
106  _keep_input_nx = true;
107  _st_ref->nx() = in_cube->size()[3];
108  } else
109  throw std::string("ERROR in stream_cube::stream_cube(): could not derive size of result cube");
110  }
