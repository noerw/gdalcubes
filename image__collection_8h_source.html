74  bounds_2d<Ta> transform(std::string srs_from, std::string srs_to) {
75  if (srs_from == srs_to) {
76  return *this;
77  }
78  OGRSpatialReference srs_in;
79  OGRSpatialReference srs_out;
80  srs_in.SetFromUserInput(srs_from.c_str());
81  srs_out.SetFromUserInput(srs_to.c_str());
82 
83  if (srs_in.IsSame(&srs_out)) {
84  return *this;
85  }
86 
87  OGRCoordinateTransformation* coord_transform = OGRCreateCoordinateTransformation(&srs_in, &srs_out);
88 
89  Ta x[4] = {left, left, right, right};
90  Ta y[4] = {top, bottom, top, bottom};
91 
92  if (coord_transform == NULL || !coord_transform->Transform(4, x, y)) {
93  throw std::string("ERROR: coordinate transformation failed (from " + srs_from + " to " + srs_to + ").");
94  }
95 
96  Ta xmin = std::numeric_limits<Ta>::is_integer ? std::numeric_limits<Ta>::max() : std::numeric_limits<Ta>::max();
97  Ta ymin = std::numeric_limits<Ta>::is_integer ? std::numeric_limits<Ta>::max() : std::numeric_limits<Ta>::max();
98  Ta xmax = std::numeric_limits<Ta>::is_integer ? std::numeric_limits<Ta>::min() : -std::numeric_limits<Ta>::max();
99  Ta ymax = std::numeric_limits<Ta>::is_integer ? std::numeric_limits<Ta>::min() : -std::numeric_limits<Ta>::max();
100  for (uint8_t k = 0; k < 4; ++k) {
101  if (x[k] < xmin) xmin = x[k];
102  if (y[k] < ymin) ymin = y[k];
103  if (x[k] > xmax) xmax = x[k];
104  if (y[k] > ymax) ymax = y[k];
105  }
106 
107  left = xmin;
108  right = xmax;
109  top = ymax;
110  bottom = ymin;
111 
112  OCTDestroyCoordinateTransformation(coord_transform);
113 
114  return *this;
115  }
